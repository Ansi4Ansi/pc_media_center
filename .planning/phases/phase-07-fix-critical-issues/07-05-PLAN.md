       import '../helpers/test_helpers.mocks.dart';
       
       void main() {
         late MockCategoryRepository mockRepository;
         late CategoryBloc bloc;
         
         setUp(() {
           mockRepository = MockCategoryRepository();
           bloc = CategoryBloc(mockRepository);
         });
         
         tearDown(() {
           bloc.close();
         });
         
         group('CategoryBloc', () {
           blocTest<CategoryBloc, CategoryState>(
             'emits [CategoryLoading, CategoryLoaded] when LoadCategories is added',
             build: () {
               when(mockRepository.getCategories())
                   .thenAnswer((_) async => [TestData.createCategory()]);
               return bloc;
             },
             act: (bloc) => bloc.add(LoadCategories()),
             expect: () => [
               isA<CategoryLoading>(),
               isA<CategoryLoaded>(),
             ],
           );
           
           blocTest<CategoryBloc, CategoryState>(
             'emits [CategoryLoading, CategoryError] when LoadCategories fails',
             build: () {
               when(mockRepository.getCategories())
                   .thenThrow(Exception('Database error'));
               return bloc;
             },
             act: (bloc) => bloc.add(LoadCategories()),
             expect: () => [
               isA<CategoryLoading>(),
               isA<CategoryError>(),
             ],
           );
           
           blocTest<CategoryBloc, CategoryState>(
             'emits [CategoryLoaded] with new category when AddCategoryEvent succeeds',
             build: () {
               final categories = [TestData.createCategory(id: 1)];
               when(mockRepository.getCategories())
                   .thenAnswer((_) async => categories);
               when(mockRepository.addCategory(any))
                   .thenAnswer((_) async {});
               return bloc;
             },
             act: (bloc) => bloc.add(AddCategoryEvent('New Category')),
             expect: () => [
               isA<CategoryLoaded>(),
             ],
             verify: (_) {
               verify(mockRepository.addCategory(any)).called(1);
               verify(mockRepository.getCategories()).called(2); // Initial + after add
             },
           );
           
           blocTest<CategoryBloc, CategoryState>(
             'emits [CategoryError] when AddCategoryEvent with empty name',
             build: () => bloc,
             act: (bloc) => bloc.add(AddCategoryEvent('')),
             expect: () => [
               isA<CategoryError>(),
             ],
           );
           
           blocTest<CategoryBloc, CategoryState>(
             'emits [CategoryLoaded] without duplicate when adding existing category',
             build: () {
               final categories = [TestData.createCategory(name: 'Existing')];
               when(mockRepository.getCategories())
                   .thenAnswer((_) async => categories);
               return bloc;
             },
             act: (bloc) => bloc.add(AddCategoryEvent('Existing')),
             expect: () => [
               isA<CategoryError>(),
             ],
           );
           
           blocTest<CategoryBloc, CategoryState>(
             'emits [CategoryLoaded] when DeleteCategoryEvent succeeds',
             build: () {
               when(mockRepository.deleteCategory(any))
                   .thenAnswer((_) async {});
               when(mockRepository.getCategories())
                   .thenAnswer((_) async => []);
               return bloc;
             },
             act: (bloc) => bloc.add(DeleteCategoryEvent(1)),
             expect: () => [
               isA<CategoryLoaded>(),
             ],
             verify: (_) {
               verify(mockRepository.deleteCategory(1)).called(1);
             },
           );
         });
       }
